CREATE TABLE ota_table_id_segment(
`id` INT(10) NOT NULL AUTO_INCREMENT COMMENT 'id',
`table_name` VARCHAR(64) UNIQUE NOT NULL COMMENT '表名',
`id_used` BIGINT NOT NULL COMMENT '不能使用的id的上界',
`step` MEDIUMINT NOT NULL COMMENT 'id号段的步长',
`update_by` BIGINT(20) NOT NULL DEFAULT '0' COMMENT '最后更新人ID',
`update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后更新时间',
 PRIMARY KEY(`id`)
)ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8 COMMENT='id号段表'
